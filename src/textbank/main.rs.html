39/// Write-ahead log record persisted as one JSON object per line.
40///
41/// Each record represents a single text insertion or update operation,
42/// containing the text ID, language code, and normalized UTF-8 text content.
43#[derive(Debug, serde::Serialize, serde::Deserialize)]
44struct WalRecord {
45    /// Unique identifier for the text entry
46    text_id: u64,
47    /// Language code (e.g., "en", "fr", "de")
48    lang: String,
49    /// Normalized UTF-8 text content
50    text: String,
51}
52
53/// Very small append-only JSON-lines write-ahead log for durability.
54///
55/// The WAL ensures data persistence by logging all text operations to disk
56/// before they're applied to the in-memory store. This provides crash recovery
57/// and data durability guarantees.
58struct Wal {
59    /// File handle for the WAL, protected by a mutex for concurrent access
60    file: Mutex<File>,
61    /// Path to the WAL file on disk
62    path: PathBuf,
63    /// When true, calls flush() after each append for stronger durability at the cost of performance
64    flush_on_write: bool,
65}
66
67impl Wal {
68    /// Opens or creates a new write-ahead log at the specified path.
69    ///
70    /// # Arguments
71    /// * `path` - The file path where the WAL should be stored
72    /// * `flush_on_write` - Whether to flush after each write for stronger durability
73    ///
74    /// # Returns
75    /// * `Result<Self>` - The WAL instance or an error if file operations fail
76    ///
77    /// # Examples
78    /// ```
79    /// let wal = Wal::open("./data/textbank.wal", false)?;
80    /// ```
81    fn open(path: impl Into<PathBuf>, flush_on_write: bool) -> Result<Self> {
82        let path = path.into();
83        // Create parent directories if they don't exist
84        std::fs::create_dir_all(path.parent().unwrap_or_else(|| std::path::Path::new(".")))?;
85
86        let file = std::fs::OpenOptions::new()
87            .create(true)
88            .append(true)
89            .read(true)
90            .open(&path)?;
91
92        Ok(Self {
93            file: Mutex::new(file),
94            path,
95            flush_on_write,
96        })
97    }
98
99    /// Appends a single record to the write-ahead log.
100    ///
101    /// The record is serialized as JSON and written to a new line in the WAL file.
102    /// If `flush_on_write` is enabled, the data is immediately flushed to disk.
103    ///
104    /// # Arguments
105    /// * `rec` - The WAL record to append
106    ///
107    /// # Returns
108    /// * `Result<()>` - Success or an error if the write operation fails
109    fn append(&self, rec: &WalRecord) -> Result<()> {
110        let mut f = self.file.lock();
111        let line = serde_json::to_string(rec)?;
112        f.write_all(line.as_bytes())?;
113        f.write_all(b"\n")?;
114
115        if self.flush_on_write {
116            f.flush()?;
117        }
118        Ok(())
119    }
120
121    /// Replays all records from the WAL file for crash recovery.
122    ///
123    /// Reads the entire WAL file line by line, deserializing each JSON record.
124    /// Empty lines are skipped to handle potential file corruption gracefully.
125    ///
126    /// # Returns
127    /// * `Result<Vec<WalRecord>>` - All records from the WAL or an error if replay fails
128    fn replay(&self) -> Result<Vec<WalRecord>> {
129        let f = std::fs::OpenOptions::new().read(true).open(&self.path)?;
130        let rdr = BufReader::new(f);
131        let mut out = Vec::new();
132
133        for line in rdr.lines() {
134            let line = line?;
135            if line.trim().is_empty() {
136                continue;
137            }
138            let rec: WalRecord = serde_json::from_str(&line)?;
139            out.push(rec);
140        }
141        Ok(out)
142    }
143}

145/// Reverse index entry to disambiguate rare hash collisions.
146///
147/// When multiple texts hash to the same value, this entry helps distinguish
148/// between them by storing the actual text content and associated ID.
149#[derive(Clone)]
150struct RevEntry {
151    /// The actual text content as a reference-counted string
152    text_arc: Arc<str>,
153    /// The unique text identifier
154    text_id: u64,
155}
156
157/// Type alias for the forward lookup index: text_id -> (language -> text_content)
158type ForwardLookup = DashMap<u64, Arc<DashMap<String, Arc<str>>>>;
159
160/// Type alias for the reverse lookup index: (language, hash) -> \[candidate_entries\]
161type ReverseLookup = DashMap<(String, u64), Arc<DashMap<u64, RevEntry>>>;
162
163/// The in-memory store providing two complementary views of the data.
164///
165/// This store maintains both forward and reverse indexes for efficient lookups:
166/// - **Forward index**: Maps text IDs to their language-specific content
167/// - **Reverse index**: Maps (language, content_hash) pairs to candidate text IDs
168///
169/// The dual-index design enables:
170/// - Fast retrieval by ID and language
171/// - Efficient deduplication using content hashes
172/// - Quick existence checks for new content
173/// - Concurrent access through lock-free data structures
174#[derive(Clone)]
175struct Store {
176    /// Forward lookup: text_id -> (lang -> string)
177    forward: Arc<ForwardLookup>,
178    /// Reverse lookup: (lang, hash) -> candidates for collision resolution
179    reverse: Arc<ReverseLookup>,
180    /// Counter for generating unique text IDs
181    next_id: Arc<parking_lot::Mutex<u64>>,
182    /// Write-ahead log for persistence
183    wal: Arc<Wal>,
184}
185
186impl Store {
187    /// Creates a new store instance with the given write-ahead log.
188    ///
189    /// # Arguments
190    /// * `wal` - The write-ahead log instance for persistence
191    ///
192    /// # Returns
193    /// * `Self` - A new store instance with empty indexes
194    fn new(wal: Wal) -> Self {
195        Self {
196            forward: Arc::new(DashMap::new()),
197            reverse: Arc::new(DashMap::new()),
198            next_id: Arc::new(parking_lot::Mutex::new(1)),
199            wal: Arc::new(wal),
200        }
201    }
202
203    /// Normalizes input text using Unicode NFC normalization.
204    ///
205    /// This ensures consistent text representation regardless of how the original
206    /// text was encoded (e.g., composed vs. decomposed Unicode characters).
207    /// Additional normalization like whitespace trimming could be added here.
208    ///
209    /// # Arguments
210    /// * `s` - Raw bytes representing UTF-8 encoded text
211    ///
212    /// # Returns
213    /// * `Result<String>` - Normalized text string or UTF-8 conversion error
214    fn normalize_text(s: &[u8]) -> Result<String> {
215        let s = std::str::from_utf8(s)?; // Treat input bytes as UTF-8
216        Ok(s.nfc().collect::<String>())
217    }
218
219    /// Computes a hash key for the reverse index from language and normalized text.
220    ///
221    /// Uses a non-ASCII separator (U+001F) to reduce the chance of accidental
222    /// collisions when concatenating language and text for hashing.
223    ///
224    /// # Arguments
225    /// * `lang` - Language code
226    /// * `norm` - Normalized text content
227    ///
228    /// # Returns
229    /// * `u64` - XXH3 hash of the combined language and text
230    fn rev_hash(lang: &str, norm: &str) -> u64 {
231        // Using a non-ASCII separator to reduce accidental collisions with concat
232        xxh3_64(format!("{}\u{001F}{}", lang, norm).as_bytes())
233    }
234
235    /// Inserts or updates a text translation with optional ID specification.
236    ///
237    /// This method handles two distinct use cases:
238    /// 1. **Update mode** (`text_id` is Some): Updates or inserts text for a specific ID
239    /// 2. **Insert mode** (`text_id` is None): Idempotent insert that returns existing ID if text already exists
240    ///
241    /// # Arguments
242    /// * `lang` - Language code for the text
243    /// * `raw` - Raw bytes representing the text content
244    /// * `text_id` - Optional text ID for update mode
245    ///
246    /// # Returns
247    /// * `Result<(u64, bool)>` - Tuple of (text_id, existed_flag) or error
248    ///   - `text_id`: The ID of the stored text
249    ///   - `existed_flag`: True if text already existed or was updated
250    fn intern_with_id(&self, lang: &str, raw: &[u8], text_id: Option<u64>) -> Result<(u64, bool)> {
251        let norm = Self::normalize_text(raw)?;
252        let h = Self::rev_hash(lang, &norm);
253
254        match text_id {
255            Some(id) => {
256                // Update path: remove old reverse entry if (id,lang) existed
257                if let Some(lm) = self.forward.get(&id) {
258                    if let Some(old) = lm.get(lang) {
259                        // Short-circuit if text is identical
260                        if old.as_ref() == norm.as_str() {
261                            return Ok((id, true));
262                        }
263                        self.reverse_remove(id, lang, &old);
264                    }
265                }
266
267                let text_arc: Arc<str> = Arc::from(norm.clone());
268
269                // Update forward index
270                let lang_map = self
271                    .forward
272                    .entry(id)
273                    .or_insert_with(|| Arc::new(DashMap::new()))
274                    .clone();
275                lang_map.insert(lang.to_string(), text_arc.clone());
276
277                // Update reverse index
278                let cand_map = self
279                    .reverse
280                    .entry((lang.to_string(), h))
281                    .or_insert_with(|| Arc::new(DashMap::new()))
282                    .clone();
283                cand_map.insert(
284                    id,
285                    RevEntry {
286                        text_arc: text_arc.clone(),
287                        text_id: id,
288                    },
289                );
290
291                // Log to WAL
292                self.wal.append(&WalRecord {
293                    text_id: id,
294                    lang: lang.to_string(),
295                    text: norm,
296                })?;
297                Ok((id, true)) // existed=true because we updated/confirmed this id
298            }
299            None => {
300                // Idempotent insert-by-(lang,text): check if content already exists
301                if let Some(bucket) = self.reverse.get(&(lang.to_string(), h)) {
302                    for entry in bucket.value().iter() {
303                        if &*entry.value().text_arc == norm.as_str() {
304                            return Ok((entry.value().text_id, true));
305                        }
306                    }
307                }
308
309                // Allocate new ID
310                let id = {
311                    let mut guard = self.next_id.lock();
312                    let id = *guard;
313                    *guard += 1;
314                    id
315                };
316
317                let text_arc: Arc<str> = Arc::from(norm.clone());
318
319                // Update forward index
320                let lang_map = self
321                    .forward
322                    .entry(id)
323                    .or_insert_with(|| Arc::new(DashMap::new()))
324                    .clone();
325                lang_map.insert(lang.to_string(), text_arc.clone());
326
327                // Update reverse index
328                let cand_map = self
329                    .reverse
330                    .entry((lang.to_string(), h))
331                    .or_insert_with(|| Arc::new(DashMap::new()))
332                    .clone();
333                cand_map.insert(
334                    id,
335                    RevEntry {
336                        text_arc: text_arc.clone(),
337                        text_id: id,
338                    },
339                );
340
341                // Log to WAL
342                self.wal.append(&WalRecord {
343                    text_id: id,
344                    lang: lang.to_string(),
345                    text: norm,
346                })?;
347                Ok((id, false))
348            }
349        }
350    }

352    /// Retrieves text content for a specific text ID and language.
353    ///
354    /// # Arguments
355    /// * `text_id` - The unique text identifier
356    /// * `lang` - The language code to retrieve
357    ///
358    /// # Returns
359    /// * `Option<Arc<str>>` - The text content if found, None otherwise
360    fn get_text(&self, text_id: u64, lang: &str) -> Option<Arc<str>> {
361        self.forward
362            .get(&text_id)
363            .and_then(|lm| lm.get(lang).map(|v| v.clone()))
364    }
365
366    /// Retrieves all language translations for a specific text ID.
367    ///
368    /// # Arguments
369    /// * `text_id` - The unique text identifier
370    ///
371    /// # Returns
372    /// * `Vec<(String, Arc<str>)>` - Vector of (language_code, text_content) pairs
373    fn get_all(&self, text_id: u64) -> Vec<(String, Arc<str>)> {
374        self.forward
375            .get(&text_id)
376            .map(|lm| {
377                lm.iter()
378                    .map(|kv| (kv.key().clone(), kv.value().clone()))
379                    .collect()
380            })
381            .unwrap_or_default()
382    }

384    /// Searches normalized text content using regular expressions.
385    ///
386    /// Supports both language-specific and cross-language search modes:
387    /// - When `lang` is specified, searches only within that language
388    /// - When `lang` is None, searches across all languages, preferring English results
389    ///
390    /// # Arguments
391    /// * `re` - Compiled regular expression to match against
392    /// * `lang` - Optional language filter
393    ///
394    /// # Returns
395    /// * `Vec<(u64, Arc<str>)>` - Vector of (text_id, matching_text) pairs
396    fn search(&self, re: &Regex, lang: Option<&str>) -> Vec<(u64, Arc<str>)> {
397        let mut out = Vec::new();
398
399        match lang {
400            Some(l) => {
401                // Language-specific search: iterate all IDs that have this language
402                for item in self.forward.iter() {
403                    let id = *item.key();
404                    if let Some(txt) = item.value().get(l) {
405                        if re.is_match(&txt) {
406                            out.push((id, txt.clone()));
407                        }
408                    }
409                }
410            }
411            None => {
412                // Cross-language search with English preference
413                use std::collections::HashSet;
414                let mut seen = HashSet::new();
415
416                for item in self.forward.iter() {
417                    let id = *item.key();
418
419                    // Prefer English if present and matches
420                    if let Some(txt) = item.value().get("en") {
421                        if re.is_match(&txt) {
422                            out.push((id, txt.clone()));
423                            seen.insert(id);
424                            continue;
425                        }
426                    }
427
428                    // Otherwise, use first matching language
429                    if !seen.contains(&id) {
430                        for kv in item.value().iter() {
431                            if re.is_match(kv.value()) {
432                                out.push((id, kv.value().clone()));
433                                break;
434                            }
435                        }
436                    }
437                }
438            }
439        }
440        out
441    }

443    /// Removes a text ID from the reverse index using the old text content.
444    ///
445    /// This is used during updates to clean up stale reverse index entries
446    /// when text content changes for an existing ID.
447    ///
448    /// # Arguments
449    /// * `text_id` - The text ID to remove from reverse index
450    /// * `lang` - The language of the text being removed
451    /// * `old_text` - The previous text content (needed to compute the hash)
452    fn reverse_remove(&self, text_id: u64, lang: &str, old_text: &str) {
453        let h = Self::rev_hash(lang, old_text);
454        if let Some(bucket) = self.reverse.get(&(lang.to_string(), h)) {
455            bucket.value().remove(&text_id);
456        }
457        // Optional: GC empty buckets to save memory
458        // if let Some(mut b) = self.reverse.get_mut(&(lang.to_string(), h)) {
459        //     if b.value().is_empty() {
460        //         self.reverse.remove(&(lang.to_string(), h));
461        //     }
462        // }
463    }

465    /// Retrieves text in a preferred language, falling back to any available language.
466    ///
467    /// Attempts to return English content first, then falls back to any available
468    /// language if English is not available.
469    ///
470    /// # Arguments
471    /// * `text_id` - The unique text identifier
472    ///
473    /// # Returns
474    /// * `Option<(String, Arc<str>)>` - Tuple of (language_code, text_content) if found
475    fn get_any(&self, text_id: u64) -> Option<(String, Arc<str>)> {
476        let lm = self.forward.get(&text_id)?;
477
478        // Prefer English if available
479        if let Some(v) = lm.get("en") {
480            return Some(("en".to_string(), v.clone()));
481        }
482
483        // Fall back to any available language
484        let result = lm
485            .iter()
486            .next()
487            .map(|kv| (kv.key().clone(), kv.value().clone()));
488        result
489    }

491    /// Loads data from the write-ahead log to rebuild both indexes.
492    ///
493    /// This method is used for crash recovery - it replays all operations from
494    /// the WAL to reconstruct the in-memory state. It also sets the next ID
495    /// counter to be one greater than the maximum ID found in the WAL.
496    ///
497    /// # Returns
498    /// * `Result<()>` - Success or error if WAL replay fails
499    fn load_from_wal(&self) -> Result<()> {
500        let mut max_id = 0u64;
501
502        for rec in self.wal.replay()? {
503            max_id = max_id.max(rec.text_id);
504            let text_arc: Arc<str> = Arc::from(rec.text.clone());
505
506            // Rebuild forward index
507            let lang_map = self
508                .forward
509                .entry(rec.text_id)
510                .or_insert_with(|| Arc::new(DashMap::new()))
511                .clone();
512            lang_map.insert(rec.lang.clone(), text_arc.clone());
513
514            // Rebuild reverse index
515            let h = Self::rev_hash(&rec.lang, &rec.text);
516            let cand_map = self
517                .reverse
518                .entry((rec.lang.clone(), h))
519                .or_insert_with(|| Arc::new(DashMap::new()))
520                .clone();
521            cand_map.insert(
522                rec.text_id,
523                RevEntry {
524                    text_arc: text_arc.clone(),
525                    text_id: rec.text_id,
526                },
527            );
528        }
529
530        // Set next ID to be one greater than the maximum found
531        *self.next_id.lock() = max_id.saturating_add(1);
532        Ok(())
533    }

536/// gRPC service implementation for the TextBank API.
537///
538/// This struct implements the `TextBank` trait generated from the protobuf
539/// definition, providing all the public API endpoints for text storage,
540/// retrieval, and search operations.
541#[derive(Clone)]
542struct Svc {
543    /// The underlying data store
544    store: Store,
545}
546
547#[tonic::async_trait]
548impl TextBank for Svc {
549    /// Inserts or updates a text translation.
550    ///
551    /// This endpoint supports two modes of operation:
552    /// - **Insert mode**: When `text_id` is 0 or unset, either allocates a new ID or returns the existing ID if the (language, text) pair already exists
553    /// - **Update mode**: When `text_id` is specified, updates or inserts the text for that specific ID
554    ///
555    /// # Arguments
556    /// * `req` - The gRPC request containing text, language, and optional text ID
557    ///
558    /// # Returns
559    /// * `Result<Response<InternReply>, Status>` - Reply with text ID and existence flag
560    ///
561    /// # Errors
562    /// * `Status::invalid_argument` - When language is empty
563    /// * `Status::internal` - When internal operations fail
564    async fn intern(&self, req: Request<InternRequest>) -> Result<Response<InternReply>, Status> {
565        let r = req.into_inner();
566
567        // Validate required parameters
568        if r.lang.is_empty() {
569            return Err(Status::invalid_argument("lang must not be empty"));
570        }
571
572        let id_opt = if r.text_id == 0 {
573            None
574        } else {
575            Some(r.text_id)
576        };
577
578        match self.store.intern_with_id(&r.lang, &r.text, id_opt) {
579            Ok((id, existed)) => Ok(Response::new(InternReply {
580                text_id: id,
581                existed,
582            })),
583            Err(e) => Err(Status::internal(format!("intern failed: {e}"))),
584        }
585    }
586
587    /// Retrieves a text translation by ID and optional language.
588    ///
589    /// When language is specified, returns the text in that specific language.
590    /// When language is empty, returns text in a preferred language (English first, then any).
591    ///
592    /// # Arguments
593    /// * `req` - The gRPC request containing text ID and optional language
594    ///
595    /// # Returns
596    /// * `Result<Response<GetReply>, Status>` - Reply with found flag and text content
597    async fn get(&self, req: Request<GetRequest>) -> Result<Response<GetReply>, Status> {
598        let r = req.into_inner();
599
600        let res = if r.lang.is_empty() {
601            self.store.get_any(r.text_id).map(|(_, s)| s)
602        } else {
603            self.store.get_text(r.text_id, &r.lang)
604        };
605
606        match res {
607            Some(s) => Ok(Response::new(GetReply {
608                found: true,
609                text: s.as_bytes().to_vec(),
610            })),
611            None => Ok(Response::new(GetReply {
612                found: false,
613                text: Vec::new(),
614            })),
615        }
616    }
617
618    /// Retrieves all language translations for a specific text ID.
619    ///
620    /// Returns all available language versions of the text content associated
621    /// with the given ID.
622    ///
623    /// # Arguments
624    /// * `req` - The gRPC request containing the text ID
625    ///
626    /// # Returns
627    /// * `Result<Response<GetAllReply>, Status>` - Reply with text ID and all translations
628    async fn get_all(&self, req: Request<GetAllRequest>) -> Result<Response<GetAllReply>, Status> {
629        let r = req.into_inner();
630
631        let translations = self
632            .store
633            .get_all(r.text_id)
634            .into_iter()
635            .map(|(lang, s)| Translation {
636                lang,
637                text: s.as_bytes().to_vec(),
638            })
639            .collect();
640
641        Ok(Response::new(GetAllReply {
642            text_id: r.text_id,
643            translations,
644        }))
645    }
646
647    /// Searches text content using regular expressions.
648    ///
649    /// Supports both language-specific and cross-language search:
650    /// - When `lang` is specified, searches only within that language
651    /// - When `lang` is empty, searches across all languages (preferring English results)
652    ///
653    /// The regex is applied to normalized (Unicode NFC) text content.
654    ///
655    /// # Arguments
656    /// * `req` - The gRPC request containing regex pattern and optional language filter
657    ///
658    /// # Returns
659    /// * `Result<Response<SearchReply>, Status>` - Reply with matching text entries
660    ///
661    /// # Errors
662    /// * `Status::invalid_argument` - When the regex pattern is invalid
663    async fn search(&self, req: Request<SearchRequest>) -> Result<Response<SearchReply>, Status> {
664        let r = req.into_inner();
665        let pattern = r.pattern;
666        let lang_opt = if r.lang.is_empty() {
667            None
668        } else {
669            Some(r.lang)
670        };
671
672        // Compile the regex pattern
673        let re = regex::Regex::new(&pattern)
674            .map_err(|e| Status::invalid_argument(format!("bad regex: {e}")))?;
675
676        let hits = self
677            .store
678            .search(&re, lang_opt.as_deref())
679            .into_iter()
680            .map(|(id, s)| SearchHit {
681                text_id: id,
682                text: s.as_bytes().to_vec(),
683            })
684            .collect();
685
686        Ok(Response::new(SearchReply { hits }))
687    }
688}

690/// Main entry point for the TextBank service.
691///
692/// Initializes the write-ahead log, creates the in-memory store, loads existing
693/// data from the WAL, and starts the gRPC server. The server runs until interrupted
694/// by a CTRL-C signal.
695#[tokio::main]
696async fn main() -> Result<()> {
697    // Server configuration
698    let addr = "0.0.0.0:50051".parse().unwrap();
